// Tahta 3x3'lük bir matristir. Her hücre: "X", "O" veya boş (" ") olabilir.
var tahta[3][3] = boş hücrelerle dolu

// Oyuncu sembolü: "X", rakip: "O"
var oyuncu = "X"
var rakip = "O"

// Oyun başlasın
print("Tic-Tac-Toe stratejik oyun başlatılıyor...")

// Her turda oyuncunun hamlesi yapılır
while oyunDevamEdiyor(tahta) do

    // 1. Oyuncu için kazanma hamlesi var mı?
    if kazanmaHamlesiVar(tahta, oyuncu) then
        yapKazanmaHamlesi(tahta, oyuncu)
        print("Oyuncu kazandı!")
        break

    // 2. Rakip kazanmak üzere mi? Engelle!
    else if kazanmaHamlesiVar(tahta, rakip) then
        engelleRakipHamlesi(tahta, rakip)
        print("Rakip engellendi.")

    // 3. Rakibin sonraki hamlesini tahmin et
    var tahminEdilenRakipHamlesi = tahminEtRakipHamlesi(tahta, rakip)

    // 4. Tahmin edilen rakip hamlesine karşı stratejik hamle yap
    if tahminEdilenRakipHamlesi != null then
        var stratejikHamle = karsilikVerHamleye(tahminEdilenRakipHamlesi)
        yapHamle(tahta, oyuncu, stratejikHamle)
        print("Tahmine dayalı stratejik hamle yapıldı.")

    // 5. Orta hücre boşsa, al!
    else if tahta[1][1] == " " then
        tahta[1][1] = oyuncu
        print("Orta hücre alındı.")

    // 6. Köşe hücrelerden biri boşsa, al!
    else if bosKoseVar(tahta) then
        alBosKose(tahta, oyuncu)
        print("Köşe hücresi alındı.")

    // 7. Kenar hücrelerden biri boşsa, al!
    else if bosKenarVar(tahta) then
        alBosKenar(tahta, oyuncu)
        print("Kenar hücresi alındı.")

    // 8. Rakip hamlesini yapar (rastgele veya stratejik)
    rakipHamlesiYap(tahta, rakip)

end while

// Oyun sonucu kontrol edilir
if kazananVar(tahta) == oyuncu then
    print("Oyun bitti: Sen kazandın!")
else if kazananVar(tahta) == rakip then
    print("Oyun bitti: Rakip kazandı.")
else
    print("Oyun bitti: Beraberlik.")
